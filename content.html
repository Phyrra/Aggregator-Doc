Finds a value matching a given condition.
Returns
(Any): Returns a found element, or null
a.find( Function | Object )
Arguments
condition (Function): A condition to be evaluated
-or-
obj (Object): An object, containing key - condition pairs
Example
a.find(function(elem) { return elem.value % 2 === 0; }); a.find({ state: 'ACTIVE', balance: function(elem) { return elem.balance > 10000; }) });
a.find( (String | Function), (Function | Any) )
Arguments
key (String): A property key, may be composited with . (e.g. state.id)
condition (Function): A condition to be evaluated
-or-
key (String): A property key, may be composited with . (e.g. state.id)
value (Any): A value to compare against
-or-
mapper (Function): A function to map the element for comparison
value (Any): A value to compare against
-or-
mapper (Function): A function to map the element for comparison
value (Any): A value to compare against
Example
a.find('state.id', function(id) { return id === 'ACTIVE'; }); a.find('state.id', 'ACTIVE'); a.find(function(elem) { return elem.state.id; }, function(id) { return id === 'ACTIVE'; }) a.find(function(elem) { return elem.state.id; } 'ACTIVE');
a.find( Array, (Function | Any) )
Arguments
mappers (Array[ String | Function ]): A list of value mappers, either composite keys or functions
condition (Function): A condition to be evaluated against all values
-or-
mappers (Array[ String | Function ]): A list of value mappers, either composite keys or functions
value (Any): A value to be compared against all values
Example
a.find(['state.id', 'employee.state.id'], function(values) { return values[0] === 'ACTIVE' && values[1] === 'ACTIVE'; }); a.find([ function(elem) { return elem.state.id; }, function(elem) { return elem.employee.state.id; } ], function(values) { return values[0] === 'ACTIVE' && values[1] === 'ACTIVE'; }); a.find(['state.id', 'employee.state.id'], 'ACTIVE');
